rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAuthenticated() {
      return request.auth != null;
    }

    function isMasterOfRoom(roomId) {
      let room = get(/databases/$(database)/documents/rooms/$(roomId)).data;
      return room != null && request.auth.uid == room.masterId;
    }

    function isRoomPlayer(roomId) {
      return exists(/databases/$(database)/documents/rooms/$(roomId)/players/$(request.auth.uid));
    }

    // Perfis de usuário
    match /users/{userId} {
      allow read: if isAuthenticated();
      allow write: if request.auth.uid == userId;
    }

    // Salas
    match /rooms/{roomId} {
      // Qualquer autenticado pode LER a sala (para verificar código ao entrar)
      allow read: if isAuthenticated();

      // Só o criador pode criar
      allow create: if isAuthenticated() &&
        request.resource.data.masterId == request.auth.uid;

      // Só o mestre pode atualizar a sala
      allow update: if isAuthenticated() && isMasterOfRoom(roomId);
      allow delete: if false;

      // Jogadores na sala
      match /players/{playerId} {
        allow read: if isAuthenticated();
        // Jogador escreve só o próprio registro, mestre escreve qualquer um
        allow write: if isAuthenticated() &&
          (request.auth.uid == playerId ||
           isMasterOfRoom(roomId));
      }
    }

    // Fichas dos jogadores
    match /sheets/{roomId}/players/{playerId} {
      allow read: if isAuthenticated() &&
        (isMasterOfRoom(roomId) ||
         request.auth.uid == playerId ||
         isRoomPlayer(roomId));

      allow write: if isAuthenticated() &&
        (request.auth.uid == playerId ||
         isMasterOfRoom(roomId));
    }

    // Histórico
    match /history/{roomId}/{sessionId} {
      allow read: if isAuthenticated() && isRoomPlayer(roomId);
      allow write: if isAuthenticated() &&
        isMasterOfRoom(roomId);
    }
  }
}
